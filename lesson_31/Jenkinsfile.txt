pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Install dependencies') {
            steps {
                bat 'python -m venv .venv'
                bat '.venv\\Scripts\\pip install -r requirements.txt'
                bat '.venv\\Scripts\\playwright install'
            }
        }

        stage('Run tests') {
            steps {
                bat '.venv\\Scripts\\pytest --junitxml=results.xml'
            }
        }

        stage('Publish results') {
            steps {
                junit 'results.xml'
            }
        }
    }
}
